func make_signed_url(path string, secret string, ttlSeconds int) (string, error) {
	// 1. Calculate Expiration Time
	expirationTime := time.Now().Add(time.Duration(ttlSeconds) * time.Second)
	expTimestamp := expirationTime.Unix()

	// 2. Construct the payload to be signed
	payload := SignaturePayload{
		Path: path,
		Exp:  expTimestamp,
	}

	payloadBytes, err := json.Marshal(payload)
	if err != nil {
		return "", fmt.Errorf("failed to marshal payload: %w", err)
	}

	// 3. Generate HMAC-SHA256 Signature
	h := hmac.New(sha256.New, []byte(secret))
	h.Write(payloadBytes)
	signatureBytes := h.Sum(nil)

	// Encode the signature to a URL-safe string
	signature := base64URL(signatureBytes)

	// 4. Construct the final URL query parameters
	params := url.Values{}
	params.Add("path", path)
	params.Add("exp", strconv.FormatInt(payload.Exp, 10))
	params.Add("sig", signature)

	// Build the final URL (assuming a base URL for demonstration)
	baseURL := "https://example.com/resource/"
	finalURL := baseURL + "?" + params.Encode()

	return finalURL, nil

import os
import boto3
from botocore.exceptions import ClientError
from typing import Dict, Any

# --- Configuration Management ---

def load_app_config() -> Dict[str, Any]:
    """
    Loads application configuration settings.

    Reads necessary configuration parameters from environment variables.
    In a real application, this might load from a structured file (YAML/JSON)
    or a dedicated configuration service.
    """
    config = {
        "aws_region": os.getenv("AWS_REGION", "us-east-1"),  # Default region if not set
        "s3_bucket_name": os.getenv("S3_BUCKET_NAME"),
        "app_secret_key": os.getenv("APP_SECRET_KEY"),  # Mandatory secret loaded from ENV
        "s3_endpoint_url": os.getenv("S3_ENDPOINT_URL"),
    }

    # Basic validation for mandatory production settings
    if not config["s3_bucket_name"]:
        raise ValueError("FATAL: S3_BUCKET_NAME environment variable is not set.")
    if not config["app_secret_key"]:
        # In production, failing to load a critical secret should halt initialization
        raise ValueError("FATAL: APP_SECRET_KEY environment variable is not set.")

    print("Configuration loaded successfully.")
    return config

# --- S3 Client Management ---

def get_s3_client() -> boto3.client:
    """
    Initializes and returns a configured boto3 S3 client for production use.

    The client configuration relies on settings loaded via load_app_config(),
    which securely reads secrets from the environment.

    Returns:
        boto3.client: A configured S3 client instance.

    Raises:
        ValueError: If critical configuration variables are missing.
        ClientError: If Boto3 fails to initialize the client (e.g., permission denied).
    """
    try:
        # 1. Load Configuration
        config = load_app_config()

        # 2. Configure and Initialize Client
        s3_client = boto3.client(
            's3',
            region_name=config["aws_region"],
            endpoint_url=config["s3_endpoint_url"],
            # Credentials are automatically inferred from environment variables,
            # IAM roles, or ~/.aws/credentials file.
        )
        
        # Optional: Test connection (best practice for production initialization)
        print(f"S3 Client initialized successfully for region: {config['aws_region']}")
        return s3_client

    except ValueError as e:
        # Handles configuration errors (missing environment variables)
        print(f"Configuration Error: {e}")
        # In a production environment, this should typically trigger a fatal application exit
        raise
